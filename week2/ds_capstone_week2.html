N-grams

buildNGrams <- function(data, n) {
  startTime <- Sys.time()
  result <- data %>% 
    unnest_tokens(ngram, text, token = "ngrams", n = n) %>%
    dplyr::count(ngram, sort = TRUE)
  elapsedTime <- Sys.time() - startTime
  list(result, elapsedTime)
}
